# One-liner to generate samples.yaml from a directory of CRAM files.
# Adjust the path glob to match your input file locations.
# The first sed removes the .cram extension for the sample name key;
# the second sed prepends the required 'samples:' YAML header.

ls /path/to/your/crams/*.cram | awk -F "/" '{print "  " $NF ": " $0}' | sed 's/\.cram//' | sed '1s/^/samples:\n/' > samples.yaml
